const clockEl     = document.getElementById('clock');
const countdownEl = document.getElementById('countdown');
const noticeEl    = document.getElementById('notice');

function updateClock() {
  const now = new Date();
  const h = now.getHours(), m = now.getMinutes(), s = now.getSeconds();
  clockEl.textContent = String(h).padStart(2,'0')+':'+String(m).padStart(2,'0')+':'+String(s).padStart(2,'0');
  countdownEl.textContent = String(59-s).padStart(2,'0');
  const totalSec = h*3600+m*60+s;
  if (totalSec < 8*3600+50*60) {
    noticeEl.textContent = '아직 09시 사냥터 개장 전입니다.';
    noticeEl.className = 'timer-notice before';
  } else {
    noticeEl.textContent = '09시 사냥터 개장 직전입니다.';
    noticeEl.className = 'timer-notice soon';
  }
}
setInterval(updateClock, 1000);
updateClock();

const pctBtns   = document.querySelectorAll('.btn-pct');
const amountInput = document.getElementById('amountInput');
const execBtns  = [document.getElementById('btnReserve'), document.getElementById('btnBuy'), document.getElementById('btnSell')];

function togglePct(el) {
  const isActive = el.classList.contains('active');
  pctBtns.forEach(b => b.classList.remove('active'));
  amountInput.classList.remove('active');
  amountInput.value = '0';
  if (!isActive) el.classList.add('active');

function activateInput() {
  pctBtns.forEach(b => b.classList.remove('active'));
  amountInput.classList.add('active');
  amountInput.select();

  if (amountInput.value.trim() !== '' && amountInput.value !== '0') amountInput.classList.add('active');
});
amountInput.addEventListener('blur', () => {
  if (!amountInput.value || amountInput.value === '0') {
    amountInput.classList.remove('active');
    amountInput.value = '0';
    execBtns.forEach(b => b.classList.remove('active'));
  }
});
function toggleExec(el) {
  const isActive = el.classList.contains('active');
  execBtns.forEach(b => b.classList.remove('active'));
  if (!isActive) {
    const pctActive = [...pctBtns].some(b => b.classList.contains('active'));
    const inputActive = amountInput.classList.contains('active');
    if (pctActive || inputActive) el.classList.add('active');
  } else {
    pctBtns.forEach(b => b.classList.remove('active'));
    amountInput.classList.remove('active');
    amountInput.value = '0';
  }
}

/* ---- 전일대비 TOP20 업비트 연동 ---- */
let coinData = [], krwNameMap = {}, ws = null;
const listEl = document.getElementById('coinList');

function renderList() {
  const sorted = [...coinData].sort((a,b) => b.change - a.change).slice(0, 20);
  listEl.innerHTML = '';
  sorted.forEach((coin, idx) => {
    const rank = idx+1;
    const rankClass = rank===1?'rank-1':rank===2?'rank-2':rank===3?'rank-3':'rank-other';
    const pct = (coin.change>=0?'+':'')+coin.change.toFixed(2)+'%';
    const name = krwNameMap[coin.code] || coin.code;
    const row = document.createElement('div');
    row.className = `coin-row ${rankClass}`;
    row.innerHTML = `<span class="coin-rank">${rank}</span><span class="coin-name">${name}</span><span class="coin-pct">${pct}</span>`;
    row.addEventListener('click', () => {
      const sym = document.getElementById('activeSymbol');
      if (sym) sym.textContent = coin.code+'/KRW';
    });
    listEl.appendChild(row);
  });
}

function connectUpbit() {
  fetch('https://api.upbit.com/v1/market/all?isDetails=false')
    .then(r => r.json())
    .then(markets => {
      const krw = markets.filter(m => m.market.startsWith('KRW-'));
      krw.forEach(m => { krwNameMap[m.market.replace('KRW-','')] = m.korean_name; });
      return fetch(`https://api.upbit.com/v1/ticker?markets=${krw.map(m=>m.market).join(',')}`);
    })
    .then(r => r.json())
    .then(tickers => {
      coinData = tickers.map(t => ({ code: t.market.replace('KRW-',''), change: t.signed_change_rate*100 }));
      renderList();
      if (ws) ws.close();
      ws = new WebSocket('wss://api.upbit.com/websocket/v1');
      ws.onopen = () => {
        ws.send(JSON.stringify([{ticket:'mobydick-top20'},{type:'ticker',codes:coinData.map(c=>'KRW-'+c.code)}]));
      };
      ws.onmessage = async (e) => {
        const data = JSON.parse(new TextDecoder().decode(await e.data.arrayBuffer()));
        if (data.type==='ticker') {
          const code = data.code.replace('KRW-','');
          const idx = coinData.findIndex(c=>c.code===code);
          if (idx>=0) coinData[idx].change = data.signed_change_rate*100;
          else coinData.push({code, change: data.signed_change_rate*100});
          renderList();
        }
      };
      ws.onclose = () => setTimeout(connectUpbit, 5000);
    })
    .catch(() => setTimeout(connectUpbit, 5000));
}
connectUpbit();

function sendChat() {

  const msg = input.value.trim();
  if (!msg) return;
  const chatEl = document.getElementById('chatMessages');
  const div = document.createElement('div');
  div.className = 'chat-msg';
  div.innerHTML = `<span class="sender" style="color:#f0b429;">👤사용자</span>${msg}`;
  chatEl.appendChild(div);
  chatEl.scrollTop = chatEl.scrollHeight;
  input.value = '';
}
