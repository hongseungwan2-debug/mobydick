let shieldActive = false;
let shieldWs = null;

function startShieldMonitor() {
  const market = 'KRW-' + (selectedCoin || 'BTC');
  shieldCancelled = false;
  resetShieldData();
  if (shieldWs) { shieldWs.close(); shieldWs = null; }
  shieldWs = new WebSocket('wss://api.upbit.com/websocket/v1');
  shieldWs.onopen = () => {
    shieldWs.send(JSON.stringify([
      { ticket: 'shield-engine' },
      { type: 'ticker',    codes: [market] },
      { type: 'orderbook', codes: [market] },
      { type: 'trade',     codes: [market] }
    ]));
    reportRow1('⚔️ 방어막 감시 시작 - ticker/orderbook/trade 연결완료');
  };
  shieldWs.onmessage = async (e) => {
    try {
      const data = JSON.parse(new TextDecoder().decode(await e.data.arrayBuffer()));
      processShieldData(data);
    } catch(err) {}
  };
  shieldWs.onclose = () => {
    if (shieldActive) setTimeout(startShieldMonitor, 3000);
  };
}

function stopShieldMonitor() {
  if (shieldWs) { shieldWs.close(); shieldWs = null; }
  resetShieldData();
}

function resetShieldData() {
  shieldData = { volumeDrop:false, upperWick:false, sellWallSurge:false, tradeStrengthDrop:false, pumpDump:false };
  shieldTriggerCount = 0;
  baseVolume = 0; candleOpen = 0; candleHigh = 0; prevTradeStrength = 50;
}

function processShieldData(data) {
  if (!shieldActive || shieldCancelled) return;
  const now = new Date();
  const h = now.getHours(), m = now.getMinutes(), s = now.getSeconds();
  if (!(h === 9 && m === 0)) return;

  if (data.type === 'ticker') {
    const price       = data.trade_price;
    const openPrice   = data.opening_price;
    const highPrice   = data.high_price;
    const accVolume   = data.acc_trade_volume;
    const tradeStr    = data.trade_strength || 50;
    if (candleOpen === 0) candleOpen = openPrice;
    if (highPrice > candleHigh) candleHigh = highPrice;
    if (s < 15 && baseVolume === 0) baseVolume = accVolume;
    if (baseVolume > 0 && s > 20) {
      const volRate = accVolume / (baseVolume * (s/15));
      if (volRate < 0.4 && !shieldData.volumeDrop) {
        shieldData.volumeDrop = true;
        reportRow1('⚠️ 거래량 급감 감지! (' + s + '초)');
        checkShieldTrigger();
      }
    }
    if (candleHigh > 0 && price > 0) {
      const wickRatio = (candleHigh - price) / candleHigh * 100;
      if (wickRatio > 3 && !shieldData.upperWick) {
        shieldData.upperWick = true;
        reportRow1('⚠️ 윗꼬리 과도 감지! (고점대비 -' + wickRatio.toFixed(1) + '%)');
        checkShieldTrigger();
      }
    }
    if (tradeStr < 40 && prevTradeStrength >= 40 && !shieldData.tradeStrengthDrop) {
      shieldData.tradeStrengthDrop = true;
      reportRow1('⚠️ 체결강도 급락 감지! (' + tradeStr.toFixed(1) + ')');
      checkShieldTrigger();
    }
    prevTradeStrength = tradeStr;
    if (candleHigh > 0 && candleOpen > 0) {
      const pumpRate = (candleHigh - candleOpen) / candleOpen * 100;
      const dumpRate = (candleHigh - price) / candleHigh * 100;
      if (pumpRate > 3 && dumpRate > 2 && !shieldData.pumpDump) {
        shieldData.pumpDump = true;
        reportRow1('⚠️ 급등 후 급락 패턴 감지! (펌핑:+' + pumpRate.toFixed(1) + '% → -' + dumpRate.toFixed(1) + '%)');
        checkShieldTrigger();
      }
    }
  }
  if (data.type === 'orderbook') {
    const askSize = data.orderbook_units ? data.orderbook_units.reduce((a,u) => a + u.ask_size, 0) : 0;
    const bidSize = data.orderbook_units ? data.orderbook_units.reduce((a,u) => a + u.bid_size, 0) : 0;
    if (bidSize > 0 && (askSize / bidSize) > 2 && !shieldData.sellWallSurge) {
      shieldData.sellWallSurge = true;
      reportRow1('⚠️ 매도호가 벽 급증 감지! (매도/매수 비율: ' + (askSize/bidSize).toFixed(1) + 'x)');
      checkShieldTrigger();
    }
  }
}

function checkShieldTrigger() {
  const count = Object.values(shieldData).filter(v => v).length;
  const el = document.getElementById('engineShield');
  if (count === 1) {
    el.classList.remove('active','danger');
    el.classList.add('warning');
  }
  if (count >= 2 && !shieldCancelled) {
    shieldCancelled = true;
    el.classList.remove('active','warning');
    el.classList.add('danger');
    document.getElementById('btnReserve').classList.remove('active');
    reportRow1('🚨 양아치 판정! ' + count + '개 신호 동시 발생 → 매수예약 자동취소 ❌');
    addChatMsg('⚔️방어막', '양아치 세력 감지! 매수예약 취소 완료. 다음 기회를 노리십시오.');
  }
}

    ]));
    reportRow1('🪂 비상탈출 감시 시작 - ticker/trade 연결완료');
  };
  escapeWs.onmessage = async (e) => {
    try {
      const data = JSON.parse(new TextDecoder().decode(await e.data.arrayBuffer()));
      processEscapeData(data);
    } catch(err) {}
  };
  escapeWs.onclose = () => {
    if (escapeActive) setTimeout(startEscapeMonitor, 3000);
  };
}

function stopEscapeMonitor() {
  if (escapeWs) { escapeWs.close(); escapeWs = null; }
  resetEscapeData();
}

function resetEscapeData() {
  escapeData = { bearishDrop:false, firstCandleBreakFail:false, tradeStrengthReverse:false, buyPowerDrop:false, volumeDecrease:false, secondCandleBreakFail:false };
  firstCandleClose = 0; secondCandleClose = 0; currentBonIdx = 0;
  escapeEntryPrice = 0; prevBuyVolume = 0; prevTotalVolume = 0; escapePrevTradeStr = 50;
}

function processEscapeData(data) {
  if (!escapeActive) return;
  const now = new Date();
  const h = now.getHours(), m = now.getMinutes(), s = now.getSeconds();
  if (h !== 9 || (m !== 1 && m !== 2)) return;

  if (data.type === 'ticker') {
    const price      = data.trade_price;
    const openPrice  = data.opening_price;
    const tradeStr   = data.trade_strength || 50;
    const accVolume  = data.acc_trade_volume;
    if (m === 1) currentBonIdx = 2;
    if (m === 2) currentBonIdx = 3;
    if (m === 1 && s < 3 && firstCandleClose === 0) {
      firstCandleClose = openPrice;
      reportRow1('🪂 2봉 감시 시작 - 1봉 종가 기준: ' + firstCandleClose.toLocaleString() + '원');
    }
    if (m === 2 && s < 3 && secondCandleClose === 0) {
      secondCandleClose = openPrice;
      reportRow1('🪂 3봉 감시 시작 - 2봉 종가 기준: ' + secondCandleClose.toLocaleString() + '원');
    }
    if (openPrice > 0 && price < openPrice * 0.99 && !escapeData.bearishDrop) {
      escapeData.bearishDrop = true;
      reportRow1('🚨 음봉 급하강 감지! 즉시 손절 준비!');
      triggerEscapeAlert('핵심');
    }
    if (currentBonIdx === 2 && firstCandleClose > 0 && price < firstCandleClose && !escapeData.firstCandleBreakFail) {
      escapeData.firstCandleBreakFail = true;
      reportRow1('🚨 1봉 종가 돌파 실패! 상승 동력 없음 → 즉시 손절 준비!');
      triggerEscapeAlert('핵심');
    }
    if (tradeStr < 50 && escapePrevTradeStr >= 50 && !escapeData.tradeStrengthReverse) {
      escapeData.tradeStrengthReverse = true;
      reportRow1('🚨 체결강도 역전! 매도세 우위 → 즉시 손절 준비!');
      triggerEscapeAlert('핵심');
    }
    escapePrevTradeStr = tradeStr;
    if (currentBonIdx === 3 && secondCandleClose > 0 && price < secondCandleClose && !escapeData.secondCandleBreakFail) {
      escapeData.secondCandleBreakFail = true;
      reportRow1('🚨 2봉 종가 돌파 실패! 3봉 홀딩 불가 → 즉시 손절 준비!');
      triggerEscapeAlert('핵심');
    }
    if (prevTotalVolume > 0 && accVolume < prevTotalVolume * 0.7 && !escapeData.volumeDecrease) {
      escapeData.volumeDecrease = true;
      reportRow1('⚠️ 거래량 감소 감지 (전조신호)');
      checkEscapeWarning();
    }
    prevTotalVolume = accVolume;
  }
  if (data.type === 'trade') {
    const askBid = data.ask_bid;
    if (askBid === 'ASK') {
      prevBuyVolume = Math.max(0, prevBuyVolume - 1);
    } else {
      prevBuyVolume++;
    }
    if (prevBuyVolume < 3 && !escapeData.buyPowerDrop) {
      escapeData.buyPowerDrop = true;
      reportRow1('⚠️ 매수세 감소 감지 (전조신호)');
      checkEscapeWarning();
    }
  }
}

function triggerEscapeAlert(type) {
  const el = document.getElementById('engineEscape');
  el.classList.remove('active','warning');
  el.classList.add('danger');
  addChatMsg('🪂비상탈출', '🚨 핵심신호 감지! 손절 버튼을 즉시 실행하십시오!');
}

function checkEscapeWarning() {
  const warningCount = [escapeData.buyPowerDrop, escapeData.volumeDecrease].filter(v=>v).length;
  if (warningCount >= 1) {
    const el = document.getElementById('engineEscape');
    if (!el.classList.contains('danger')) {
      el.classList.remove('active');
      el.classList.add('warning');
      if (warningCount === 2) {
        reportRow1('⚠️ 전조신호 2개 동시 발생! 손절 준비 대기!');
        addChatMsg('🪂비상탈출', '⚠️ 전조신호 감지. 손절 버튼 준비하십시오.');
      }
    }
  }
}

let coinData = [], krwNameMap = {}, ws = null;
let selectedCoin = 'BTC';
const listEl = document.getElementById('coinList');

function renderList() {
  listEl.innerHTML = '';
  coinData.forEach((coin, idx) => {
    const rank = idx+1;
    const rankClass = rank===1?'rank-1':rank===2?'rank-2':rank===3?'rank-3':'rank-other';
    const pct = (coin.change>=0?'+':'')+coin.change.toFixed(2)+'%';
    const name = krwNameMap[coin.code] || coin.code;
    const row = document.createElement('div');
    // 02 선택된 코인 selected 클래스 유지
    const isSelected = selectedCoin === coin.code;
    row.className = `coin-row ${rankClass}${isSelected ? ' selected' : ''}`;
    row.innerHTML = `<span class="coin-rank">${rank}</span><span class="coin-name">${name}</span><span class="coin-pct">${pct}</span>`;
    row.addEventListener('click', () => {
      // 02 기존 selected 제거 후 현재 행에 추가
      document.querySelectorAll('.coin-row').forEach(r => r.classList.remove('selected'));
      row.classList.add('selected');
      selectedCoin = coin.code;
      changeChartSymbol(coin.code);
    });
    listEl.appendChild(row);
  });
}

function connectUpbit() {
  fetch('https://api.upbit.com/v1/market/all?isDetails=false')
    .then(r => r.json())
    .then(markets => {
      const krw = markets.filter(m => m.market.startsWith('KRW-'));
      krw.forEach(m => { krwNameMap[m.market.replace('KRW-','')] = m.korean_name; });
      return fetch(`https://api.upbit.com/v1/ticker?markets=${krw.map(m=>m.market).join(',')}`);
    })
    .then(r => r.json())
    .then(tickers => {
      coinData = tickers
        .map(t => ({ code: t.market.replace('KRW-',''), change: t.signed_change_rate*100 }))
        .sort((a,b) => b.change - a.change)
        .slice(0, 20);
      renderList();
      const top20Markets = coinData.map(c => 'KRW-'+c.code);
      if (ws) ws.close();
      ws = new WebSocket('wss://api.upbit.com/websocket/v1');
      ws.onopen = () => {
        ws.send(JSON.stringify([{ticket:'mobydick-top20'},{type:'ticker',codes:top20Markets}]));
      };
      ws.onmessage = async (e) => {
        const data = JSON.parse(new TextDecoder().decode(await e.data.arrayBuffer()));
        if (data.type==='ticker') {
          const code = data.code.replace('KRW-','');
          const idx = coinData.findIndex(c=>c.code===code);
          if (idx>=0) coinData[idx].change = data.signed_change_rate*100;
          coinData.sort((a,b) => b.change - a.change);
          renderList();
          updateBuyTradeRow(data);
        }
      };
      ws.onclose = () => setTimeout(connectUpbit, 5000);
    })
    .catch(() => setTimeout(connectUpbit, 5000));
}
connectUpbit();

let buyTradeEntry  = null;
let sellTradeEntry = null;

// 매수 체결 등록
function setBuyTradeEntry(coinCode, entryPrice, entryChangePct, investAmount) {
  const now = new Date();
  const timeStr = String(now.getHours()).padStart(2,'0')+':'+String(now.getMinutes()).padStart(2,'0')+':'+String(now.getSeconds()).padStart(2,'0');
  buyTradeEntry = {
    time: timeStr,
    coinCode: coinCode,
    coinName: krwNameMap[coinCode] || coinCode,
    entryPrice: entryPrice,
    entryChangePct: entryChangePct,
    investAmount: investAmount,
  };
  document.getElementById('buyTradeTime').textContent = timeStr + ' ' + (krwNameMap[coinCode]||coinCode) + ' ' + Math.round(investAmount).toLocaleString() + '원 체결';
  reportRow1('✅ 매수체결완료 ' + timeStr);
}

// 매도 체결 등록
function setSellTradeEntry(coinCode, sellPrice, sellChangePct, sellAmount) {
  const now = new Date();
  const timeStr = String(now.getHours()).padStart(2,'0')+':'+String(now.getMinutes()).padStart(2,'0')+':'+String(now.getSeconds()).padStart(2,'0');
  sellTradeEntry = {
    time: timeStr,
    coinCode: coinCode,
    coinName: krwNameMap[coinCode] || coinCode,
    sellPrice: sellPrice,
    sellChangePct: sellChangePct,
    sellAmount: sellAmount,
  };
  document.getElementById('sellTradeTime').textContent = timeStr + ' ' + (krwNameMap[coinCode]||coinCode) + ' ' + Math.round(sellAmount).toLocaleString() + '원 체결';
  renderSellTradeRow();
  reportRow1('✅ 매도체결완료 ' + timeStr);
}

function updateBuyTradeRow(tickerData) {
  if (!buyTradeEntry) return;
  const code = tickerData.code.replace('KRW-','');
  if (code !== buyTradeEntry.coinCode) return;

  const currentPrice     = tickerData.trade_price;
  const currentChangePct = tickerData.signed_change_rate * 100;
  const myGainPct        = currentChangePct - buyTradeEntry.entryChangePct;
  const currentValue     = buyTradeEntry.investAmount * (1 + myGainPct / 100);

  const gainSign    = myGainPct >= 0 ? '+' : '';
  const curSign     = currentChangePct >= 0 ? '+' : '';
  const entrySign   = buyTradeEntry.entryChangePct >= 0 ? '+' : '';
  const gainClass   = myGainPct >= 0 ? 'buy-pos' : 'buy-neg';

  const dataEl = document.getElementById('buyTradeData');
  dataEl.innerHTML = `
    <span class="trade-item">
      <span class="trade-item-val buy-coin">${buyTradeEntry.coinName}:${buyTradeEntry.entryPrice.toLocaleString()}/<span style="color:#ffaa44">${currentPrice.toLocaleString()}</span></span>
    </span>
    <span class="trade-item">
      <span class="trade-item-label">·상승률:</span>
      <span class="trade-item-val buy-pos">${curSign}${currentChangePct.toFixed(1)}%</span>
    </span>
    <span class="trade-item">
      <span class="trade-item-label">·수익:</span>
      <span class="trade-item-val ${gainClass}">${gainSign}${myGainPct.toFixed(1)}%</span>
    </span>
    <span class="trade-item">
      <span class="trade-item-label">·매도지점:</span>
      <span class="trade-item-val buy-pos">${entrySign}${buyTradeEntry.entryChangePct.toFixed(1)}%</span>
    </span>
    <span class="trade-item">
      <span class="trade-item-label">·매수금액:</span>
      <span class="trade-item-val buy-cur">${Math.round(buyTradeEntry.investAmount).toLocaleString()}원/<span style="color:#ffaa44">${Math.round(currentValue).toLocaleString()}원</span></span>
    </span>
  `;
}

function renderSellTradeRow() {
  if (!sellTradeEntry) return;
  const dataEl = document.getElementById('sellTradeData');
  const sign = sellTradeEntry.sellChangePct >= 0 ? '+' : '';
  dataEl.innerHTML = `

    <span class="trade-item">
      <span class="trade-item-label">·매도가:</span>

    </span>
    <span class="trade-item">
      <span class="trade-item-label">·상승률:</span>
      <span class="trade-item-val sell-pos">${sign}${sellTradeEntry.sellChangePct.toFixed(1)}%</span>
    </span>
    <span class="trade-item">
      <span class="trade-item-label">·매도금액:</span>
      <span class="trade-item-val sell-cur">${Math.round(sellTradeEntry.sellAmount).toLocaleString()}원</span>
    </span>
  `;
}
